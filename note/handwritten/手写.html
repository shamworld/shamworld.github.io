Function.prototype.myCall = function(context=window,...args){
    if(typeof this!=='function'){
        throw new TypeError('not function');
    }
    //Symbol创建得不会重复 即使key一样
    var fn = Symbol('fn');
    context[fn] = this;
    var res = context[fn](...args);
    delete context[fn];
    return res;
}

class RoyPromise{
    static PENDING = 'PENDING';
    static FULFILED = 'FULFILED';
    static REJECTED = 'REJECTED';

    constructor(callback){
        if(typeof callback !== 'function'){
            throw new Error('callback is not function')
        }
        //执行状态
        this.promiseStatus = RoyPromise.PENDING;
        //自定义初始值
        this.value = null;
        // 定义resolve函数队列 reject函数队列
        this.resolveList = [];
        this.rejectList = [];

        //调用callback函数
        callback(this._resolve.bind(this),this._reject.bind(this));
    }
    _resolve(val){
        //因为promise是微任务，所以我们不能使用setTimeout，这里面我们可以使用postMessage
       window.addEventListener('message',()=>{
        if(this.promiseStatus !== RoyPromise.PENDING) return;
        this.promiseStatus = RoyPromise.FULFILED;
        this.value = val;
        let handle;
        while(handle = this.resolveList.shift()){
            handle(this.value);
        }
       });
       window.postMessage('');
    }

    _reject(val){
        window.addEventListener('message',()=>{
            if(this.promiseStatus !== RoyPromise.PENDING) return;
            this.promiseStatus = RoyPromise.REJECTED;
            this.value = val;
            let handle;
            while(handle = this.rejectList.shift()){
                handle(this.value);
            }
           });
           window.postMessage('');
    }
    then(resolveHandle,rejectHandle){
        return new RoyPromise((resolve,reject) => {
            function newResolveHandle(val){
                if (typeof resolveHandle === 'function') {
                    let result = resolveHandle(val);
                    if (result instanceof RoyPromise) {
                        result.then(resolve,reject);
                    } else {
                        resolve(result);
                    }
                } else {
                    resolve(val);
                }
            }
            function newRejectHandle(val){
                if (typeof rejectHandle === 'function') {
                    let result = rejectHandle(val);
                    if (result instanceof RoyPromise) {
                        this.then(resolve,reject);
                    } else {
                        reject(result);
                    }
                } else {
                    reject(val);
                }
            }
            this.resolveList.push(newResolveHandle);
            this.rejectList.push(newRejectHandle);
        })
    }
    catch(rejectHandle){
        return this.then(undefined,rejectHandle);
    }
    static all(promiseList){
        return new RoyPromise((resolve,reject) => {
            let arr = [];
            let index = 0;
            promiseList.forEach(promise => {
                promise.then(res=>{
                    arr.push(res);
                    index++;
                    if (index == promiseList.length) {
                        resolve(arr);
                    }
                }).catch((e)=>{
                    reject(e)
                });
            });
        })
    }
    static  race(promiseList){
        return new RoyPromise((resolve,reject)=>{
            promiseList.forEach(promise => {
                promise.then(res=>{
                    resolve(res);
                }).catch((e)=>{
                    reject(e)
                });
            })
        })
    }
    static resolve(val){
        return new RoyPromise((resolve) => {
            resolve(val);
        })
    }
    static reject(val){
        return new RoyPromise((resolve,reject) => {
            reject(val);
        })
    }
}

class ObServe{
    constructor(){
        this.events = {};
    }
    on(eventName,fn){
        this.events[eventName] = this.events[eventName] || [];
        this.events[eventName].push(fn);
    }
    emit(eventName,data){
        const fnList = this.events[eventName] || [];
        fnList.forEach(fn => {
            !fn.onece&&fn(data);
        })
    }
    remove(eventName,fn){
        if(fn){
            this.events[eventName] = this.events[eventName].filter(f => f!==fn);
        }else{
            delete this.events[eventName]
        }
    }
    onece(eventName,data){
        const fnList = this.events[eventName] || [];
        fnList.forEach(fn => {
            if(!fn.onece){
                fn.onece = true;
                fn(data);
            }
        })
    }
}

# throttle(节流)和debouce(防抖)

节流:高频时间触发js，但n秒时间内只执行一次，所以节流会稀释函数得执行频率 防抖:高频时间触发js后n秒内函数只会执行一次，如果n秒内高频时间再次触发，则重新计算时间

const throttle = (fn,time)=>{
    let isFlag = false;
    return function(){
        if(isFlag) return;
        isFlag = true;
        setTimeout(()=>{
            fn.apply(this,arguments);
            isFlag = false;
        },time);
    }
}

const debouce = (fn,time)=>{
    let timeout = null;
    return function(){
        clearTimeout(timeout);
        timeout = setTimeout(()=>{
            fn.apply(this,arguments);
        },time);
    }
}

# filter

Array.prototype.filter = function(callback,thisArgs){
    if(this==='undefined'){
        throw new TypeError('This is null or not undefined');
    }
    if(typeof callback!=='function'){
        throw new TypeError(callback+'not function');
    }

    var res = [];
    var obj = Object(this);
    var len = obj.length>>0;
    for(let i=0;i<len;i++){
        if(i in obj){
            if(callback.call(thisArgs,obj[i],i,obj)){
                res.push(obj[i]);
            }
        }
    }
    return res;
}

Array.prototype.reduce = function(callback,initValue){
    if(this==='undefined'){
        throw new ErrorType('this is null or undefined');
    }
    if(callback!=='function'){
        throw new ErrorType('not function');
    }
    var obj = Object(this);
    var len = obj.length>>0;
    var res = initValue;
    let index = 0;
    if(res===undefined){
        while(index<len&&!(index in obj)){
            index++;
        }
        if(index>=len){
            throw new ErrorType('Reduce of enty array with no initValue');
        }
        res = obj[index++];
    }

    while(index<len){
        if(index in obj){
            res = callback(undefined,res,obj[index],index,obj);
        }
        index++;
    }
    return res;
}

function myInstanceof(left,right){
    if(typeof right!=='function'){
        throw new Error('not function');
    }
    if(!left||(typeof left!=='object')||(typeof left!=='function')){
        return false;
    }
    const leftVal = left.__proto__;
    const rightVal = right.prototype;
    while(leftVal){
        if(leftVal==rightVal){
            return true;
        }
        leftVal = leftVal.__proto__;
    }
    return false;
}

class Lazy{
    constructor(){
        this.chain = Promise.resolve();
    }
    keep(time){
        return new Promise((resolve, reject) => {
            setTimeout(()=>{
                resolve()
            },time);
        })
    }
    sleep(time){
        this.chain = this.chain.then(()=>{
            return this.keep(time);
        })
        return this;

        return this;

# 手写一个深拷贝

function deepClone(obj={}){
    if (typeof obj !== 'object' || typeof obj == null) {
        return obj;
    }
    let result;
    if (Array.isArray(obj)){
        result = [];
    } else {
        result = {}
    }
    for (let key in obj) {
        //返回一个布尔值，判断对象是否包含特定的自身（非继承）属性
        if (obj.hasOwnProperty(key)){
            result[key] = deepClone(obj[key]);
        }
        
    }
    return result;
}

const curry = fn => {
    const len = fn.length;
    const judge = (...args1) => args1.length>=len?
                  fn(...args1) : (...args2) => judge(...[...args1,...args2]);
    return judge;
}

# LRU

class LRUCache {
     constructor(capacity) {
        // 最大缓存容量
        this.capacity = capacity || 10
        // 数据缓存对象
        this.cache = {}
        // 键名缓存数组，还可以提供key的访问顺序
        this.keys = []
    }
    getIdx(key) {
        return this.keys.indexOf(key)
    }
    get(key) {
        const idx = this.getIdx(key)
        if(idx === -1) return -1
        // 更新key的顺序
        this.keys.push(this.keys.splice(idx, 1)[0])
        return this.cache[key]
    }
    put(key, value) {
        const idx = this.getIdx(key)
        if(idx!==-1) {
            // 更新原有数据和访问顺序
            this.keys.push(this.keys.splice(idx, 1)[0])
        } else {
            // 超出容器
            if(this.keys.length === this.capacity) {
                this.cache[this.keys.shift()] = null
            }
            this.keys.push(key)
        }
        this.cache[key] = value
    }
    getCache() {
        return this.cache
    }
    getKeys() {
        return this.keys
    }
}
